593  template<typename IT>
594  static std::vector<Cal_Bdd>
595  C_Bdd_vector(Cal_BddManager bddManager, IT begin, IT end)
596  {
597  // TODO: tidy up with template overloading
598 
599  std::vector<Cal_Bdd> out;
600  out.reserve(std::distance(begin, end));
601 
602  while (begin != end) {
603  const typename IT::value_type &x = *(begin++);
604 
605  if constexpr (std::is_same_v<typename IT::value_type, BDD>) {
606  // TODO: assert same 'bddManager'...
607 
608  BDD::UnFree(x._bddManager, x._bdd);
609  out.push_back(x._bdd);
610  } else if constexpr (std::is_same_v<typename IT::value_type, int>) {
611  out.push_back(Cal_BddManagerGetVarWithId(bddManager, x));
612  }
613  }
614 
615  out.push_back(Cal_BddNull(bddManager));
616 
617  return std::move(out);
618  }

624  static std::vector<BDD>

629  for (int i = 0; Cal_BddIsBddNull(bddManager, bddArray[i]) == 0; i++){
