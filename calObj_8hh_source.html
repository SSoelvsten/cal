550  C_Bdd_vector(Cal_BddManager bddManager, IT begin, IT end)
551  {
552  // TODO: tidy up with template overloading
553 
554  std::vector<Cal_Bdd> out;
555  out.reserve(std::distance(begin, end));
556 
557  while (begin != end) {
558  const typename IT::value_type &x = *(begin++);
559 
560  if constexpr (std::is_same_v<typename IT::value_type, BDD>) {
561  // TODO: assert same 'bddManager'...
562 
563  BDD::UnFree(x._bddManager, x._bdd);
564  out.push_back(x._bdd);
565  } else if constexpr (std::is_same_v<typename IT::value_type, int>) {
566  out.push_back(Cal_BddManagerGetVarWithId(bddManager, x));
567  }
568  }
569 
570  out.push_back(Cal_BddNull(bddManager));
571 
572  return std::move(out);
573  }
